# install openconnect

- sudo apt install openconnect

# add vpn to your sudo command

- cd to vpn directory
- sudo cp vpn /usr/bin

# Config VPN
(doesn't need every time)
- sudo vpn config
- chose your config choice

# Start VPN

- $sudo vpn start

# Stop vpn

- $sudo vpn stop
